Supporting Reads Distribution

library(scales)
library(ggplot2)
#SRR2082439
mbonenovel = read.csv("../myoblast_myotube/GSE70389/star/SRR2082439/novel_junctions_0multi.tsv", header=FALSE, sep = "\t")
mboneannotated = read.csv("../myoblast_myotube/GSE70389/star/SRR2082439/annotated_junctions_0multi.tsv", header=FALSE, sep="\t")
mbonenovel$junctions = 'novel'
mboneannotated$junctions = 'annotated'
mbonereads = rbind.data.frame(mbonenovel, mboneannotated)
myoblast1 = ggplot(mbonereads, aes(mbonereads$V7, fill = junctions)) + geom_histogram(alpha = 0.5, position = 'identity', binwidth = 0.1) + scale_x_continuous(name="Number of reads mapped to junction", breaks = c(1,2,3,5,10,20,100,500,1000,3000), trans="log10", limits=c(0.7,3000)) + ylab("Number of junctions") + ggtitle("Reads supporting splice junctions (SRR2082439) ") + theme(panel.grid.major = element_blank(), panel.grid.minor = element_blank(),
panel.background = element_blank(), axis.line = element_line(colour = "black"))
myoblast1

#Do it for intron size too
novelintron = read.csv("../myoblast_myotube/GSE70389/star/SRR2082439/novel_reads_v_intronlength.tsv", sep="\t", header=TRUE)
annotintron = read.csv("../myoblast_myotube/GSE70389/star/SRR2082439/annotated_reads_v_intronlength.tsv", sep="\t", header=TRUE)
novelintron$junctions = 'novel'
annotintron$junctions = 'annotated'
introns = rbind.data.frame(novelintron, annotintron)
#Get rid of 0's
introndist = ggplot(introns, aes(introns$Intron_length, fill = junctions)) +
  geom_histogram(alpha = 0.5, position = 'identity', binwidth = 0.05) +
  scale_x_continuous(name="Intron Length (bp)", breaks = c(1,2,3,5,10,20,60,200,2000, 20000,200000), trans="log10", labels = comma,limits=c(10,200000)) + 
  ylab("Frequency") + 
  ggtitle("Intron length distribution (SRR2082439) ") + 
  geom_vline(xintercept = 59, linetype="longdash") + 
  annotate("text", x=35, y=1000, label="<60nt", size=4) + 
  theme(panel.grid.major = element_blank(), panel.grid.minor = element_blank())
introndist
